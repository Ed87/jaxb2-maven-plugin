105         unitUnderTest.addSourceFiles(sourceFiles);
106         final SearchableDocumentation result = unitUnderTest.process();
107 
108         // Assert
109         final SortedMap<String, SortableLocation> path2locMap = new TreeMap<String, SortableLocation>();
110 
111         for (String currentPath : result.getPaths()) {
112             path2locMap.put(currentPath, result.getLocation(currentPath));
113         }
114 
115         final String prefix = "testdata.schemageneration.javadoc.annotated.";
116         final String fieldAccessPrefix = prefix + "AnnotatedXmlNameAnnotatedClassWithFieldAccessTypeName#";
117         final String methodAccessPrefix = prefix + "AnnotatedXmlNameAnnotatedClassWithMethodAccessTypeName#";
118 
119         // First, check the field-annotated class.
120         final SortableLocation stringFieldLocation = result.getLocation(fieldAccessPrefix + "annotatedStringField");
121         final SortableLocation integerFieldLocation = result.getLocation(fieldAccessPrefix + "annotatedIntegerField");
122         final SortableLocation stringMethodLocation = result.getLocation(fieldAccessPrefix + "getStringField()");

133         Assert.assertEquals("annotatedIntegerField", integerFieldLocation.getAnnotationRenamedTo());
134 
135         Assert.assertEquals(JavaDocData.NO_COMMENT, result.getJavaDoc(stringMethodLocation.getPath()).getComment());

151         Assert.assertTrue(integerMethodLocation2 instanceof MethodLocation);
152

154         Assert.assertNull(integerFieldLocation2.getAnnotationRenamedTo());

156         Assert.assertEquals("annotatedIntegerMethod", integerMethodLocation2.getAnnotationRenamedTo());

168     @Test
169     public void validateExtractEnumJavaDocs() {
170 
171         // Assemble
172         final JavaDocExtractor unitUnderTest = new JavaDocExtractor(log);
173         unitUnderTest.setEncoding("UTF-8");
174 
175         final List<File> sourceDirs = Collections.singletonList(javaDocEnumsDir);
176         final List<Filter<File>> excludeFilesMatching = new ArrayList<Filter<File>>();
177         excludeFilesMatching.add(new PatternFileFilter(Collections.singletonList("\\.xsd")));
178         Filters.initialize(log, excludeFilesMatching);
179 
180         final List<File> sourceFiles = FileSystemUtilities.resolveRecursively(sourceDirs, excludeFilesMatching, log);
181         Assert.assertEquals(2, sourceFiles.size());
182 
183         // Act
184         unitUnderTest.addSourceFiles(sourceFiles);
185         final SearchableDocumentation result = unitUnderTest.process();
186 
187         final SortedMap<SortableLocation, JavaDocData> sortableLocations2JavaDocDataMap = result.getAll();
188         final SortedMap<String, SortableLocation> path2LocationMap = new TreeMap<String, SortableLocation>();
189 
190         for (Map.Entry<SortableLocation, JavaDocData> current : sortableLocations2JavaDocDataMap.entrySet()) {
191             path2LocationMap.put(current.getKey().getPath(), current.getKey());
192         }
193 
194         // Assert
195         Assert.assertEquals(16, sortableLocations2JavaDocDataMap.size());
196 
197         final List<String> paths = Arrays.asList(
198                 "enums",
199                 "enums.AmericanCoin",
200                 "enums.AmericanCoin#1",
201                 "enums.AmericanCoin#5",
202                 "enums.AmericanCoin#10",
203                 "enums.AmericanCoin#25",
204                 "enums.AmericanCoin#getValue()",
205                 "enums.AmericanCoin#value",
206                 "enums.FoodPreference",
207                 "enums.FoodPreference#LACTO_VEGETARIAN",
208                 "enums.FoodPreference#NONE",
209                 "enums.FoodPreference#VEGAN",
210                 "enums.FoodPreference#isMeatEater()",
211                 "enums.FoodPreference#isMilkDrinker()",
212                 "enums.FoodPreference#meatEater",
213                 "enums.FoodPreference#milkDrinker");
214         for (String current : paths) {
215             Assert.assertTrue("Required path [" + current + "] not found.",
216                     path2LocationMap.keySet().contains(current.trim()));
217         }
218 
219         final SortableLocation enumClassLocation = path2LocationMap.get("enums.FoodPreference");
220         final JavaDocData enumClassJavaDocData = sortableLocations2JavaDocDataMap.get(enumClassLocation);
221 
222         Assert.assertEquals("Simple enumeration example defining some Food preferences.",
223                 enumClassJavaDocData.getComment());
224         Assert.assertEquals("Vegan who will neither eat meats nor drink milk.",
225                 sortableLocations2JavaDocDataMap.get(path2LocationMap.get("enums.FoodPreference#VEGAN")).getComment());
226     }

252     @Test
253     public void validateJavaDocDataFromProcessing() {
254 
255         // Assemble
256         final String basicPackagePath = "basic";
257         final String nodeProcessorClassPath = "basic.NodeProcessor";
258         final String acceptMethodPath = "basic.NodeProcessor#accept(org.w3c.dom.Node)";
259         final String processMethodPath = "basic.NodeProcessor#process(org.w3c.dom.Node)";
260 
261         final JavaDocExtractor unitUnderTest = new JavaDocExtractor(log);
262         final List<File> sourceDirs = Arrays.<File>asList(javaDocBasicDir);
263         final List<File> sourceFiles = FileSystemUtilities.resolveRecursively(sourceDirs, null, log);
264 
265         // Act
266         unitUnderTest.addSourceFiles(sourceFiles);
267         final SearchableDocumentation result = unitUnderTest.process();
268 
269         // Assert
270         /*
271          +=================
272          | Comment:
273          | No JavaDoc tags.
274          +=================
275          */
276         final SortableLocation packageLocation = result.getLocation(basicPackagePath);
277         final JavaDocData basicPackageJavaDoc = result.getJavaDoc(basicPackagePath);
278         Assert.assertTrue(packageLocation instanceof PackageLocation);
279 
280         final PackageLocation castPackageLocation = (PackageLocation) packageLocation;
281         Assert.assertEquals("basic", castPackageLocation.getPackageName());
282         Assert.assertEquals(JavaDocData.NO_COMMENT, basicPackageJavaDoc.getComment());
283         Assert.assertEquals(0, basicPackageJavaDoc.getTag2ValueMap().size());
284 
285         /*
286          +=================
287          | Comment: Processor/visitor pattern specification for DOM Nodes.
288          | 2 JavaDoc tags ...
289          | author: <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, Mr. Foo
290          | see: org.w3c.dom.Node
291          +=================
292          */
293         final SortableLocation classLocation = result.getLocation(nodeProcessorClassPath);
294         final JavaDocData nodeProcessorClassJavaDoc = result.getJavaDoc(nodeProcessorClassPath);
295         Assert.assertTrue(classLocation instanceof ClassLocation);
296 
297         final ClassLocation castClassLocation = (ClassLocation) classLocation;
298         Assert.assertEquals("basic", castClassLocation.getPackageName());
299         Assert.assertEquals("NodeProcessor", castClassLocation.getClassName());
300         Assert.assertEquals("Processor/visitor pattern specification for DOM Nodes.",
301                 nodeProcessorClassJavaDoc.getComment());
302 
303 
304         final SortedMap<String, String> classTag2ValueMap = nodeProcessorClassJavaDoc.getTag2ValueMap();
305         Assert.assertEquals(2, classTag2ValueMap.size());
306         Assert.assertEquals("org.w3c.dom.Node", classTag2ValueMap.get("see"));
307         Assert.assertEquals("<a href=\"mailto:lj@jguru.se\">Lennart J&ouml;relid</a>, Mr. Foo",
308                 classTag2ValueMap.get("author"));
309 
310         /*
311          +=================
312          | Comment: Defines if this visitor should process the provided node.
313          | 2 JavaDoc tags ...
314          | param: aNode The DOM node to process.
315          | return: <code>true</code> if the provided Node should be processed by this NodeProcessor.
316          +=================
317          */
318         final SortableLocation acceptMethodLocation = result.getLocation(acceptMethodPath);
319         final JavaDocData acceptMethodClassJavaDoc = result.getJavaDoc(acceptMethodPath);
320         Assert.assertTrue(acceptMethodLocation instanceof MethodLocation);
321 
322         final MethodLocation castMethodLocation = (MethodLocation) acceptMethodLocation;
323         Assert.assertEquals("basic", castMethodLocation.getPackageName());
324         Assert.assertEquals("NodeProcessor", castMethodLocation.getClassName());
325         Assert.assertEquals("(org.w3c.dom.Node)", castMethodLocation.getParametersAsString());
326         Assert.assertEquals("Defines if this visitor should process the provided node.",
327                 acceptMethodClassJavaDoc.getComment());
328 
329         final SortedMap<String, String> methodTag2ValueMap = acceptMethodClassJavaDoc.getTag2ValueMap();
330         Assert.assertEquals(2, methodTag2ValueMap.size());
331         Assert.assertEquals("aNode The DOM node to process.", methodTag2ValueMap.get("param"));
332         Assert.assertEquals("<code>true</code> if the provided Node should be processed by this NodeProcessor.",
333                 methodTag2ValueMap.get("return"));
334     }
